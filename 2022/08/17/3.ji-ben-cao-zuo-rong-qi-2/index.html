0.安装Docker

Docker 分为 CE 和 EE 两大版本。CE 即社区版（免费，支持周期 7 个月），EE 即企业版，强调安全，付费使用，支持周期 24 个月。

Docker CE 分为 stable test 和 nightly 三个更新频道。

官方网站上有各种环境下的 安装指南，这里主要介绍 Docker CE 在 CentOS上的安装。

0.1.CentOS7安装Docker

Docker CE 支持 64 位版本 CentOS 7，并且要求内核版本不低于 3.10， CentOS 7 满足最低内核的要求，所以我们在CentOS 7安装Docker。

0.1.1.卸载（可选）

如果之前安装过旧版本的Docker，可以使用下面命令卸载：

yum remove docker \
                  docker-client \
                  docker-client-latest \
                  docker-common \
                  docker-latest \
                  docker-latest-logrotate \
                  docker-logrotate \
                  docker-selinux \
                  docker-engine-selinux \
                  docker-engine \
                  docker-ce

0.1.2.安装docker

首先需要大家虚拟机联网，安装yum工具

yum install -y yum-utils \
           device-mapper-persistent-data \
           lvm2 --skip-broken

然后更新本地镜像源：

# 设置docker镜像源
yum-config-manager \
    --add-repo \
    https://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo
    
sed -i 's/download.docker.com/mirrors.aliyun.com\/docker-ce/g' /etc/yum.repos.d/docker-ce.repo

yum makecache fast

然后输入命令：

yum install -y docker-ce

docker-ce为社区免费版本。稍等片刻，docker即可安装成功。

0.1.3.启动docker

Docker应用需要用到各种端口，逐一去修改防火墙设置。非常麻烦，因此建议大家直接关闭防火墙！

启动docker前，一定要关闭防火墙后！！

# 关闭
systemctl stop firewalld
# 禁止开机启动防火墙
systemctl disable firewalld

通过命令启动docker：

systemctl start docker  # 启动docker服务

systemctl stop docker  # 停止docker服务

systemctl restart docker  # 重启docker服务

systemctl status docker #查看docker是否启动

然后输入命令，可以查看docker版本：

docker -v

1.容器命令

1.1.启动

docker run -p 3306:3306 --name mysql \
-v /mydata/mysql/log:/var/log/mysql \
-v /mydata/mysql/data:/var/lib/mysql\
-v /mydata/mysql/conf:/etc/mysql \
-e MYSQL_ROOT_PASSWORD=123456 \
-d mysql:5.7



#每个docker容器里面其实就是一个linux系统
#通过 docker exce -it 可以进入容器中的 /mydata/mysql/log 查看日志等，但是为了方便修改，所以有了文件映射，也就是 -v参数
#不建议在docker中部署有状态服务，列如redis，mysql。因为容器一但删除，将会造成数据丢失。所以一定要做好容器卷映射进行备份
-v 文件映射
-e 设置容器内部属性
-d 后台运行
--name 取名
-p 端口映射
--privileged=true #让容器内的root拥有真正的root权限
--volumes-from 容器名 #继承另一个容器的映射文件，两个容器共享使用同一份映射文件。数据互相传递

1.2.常用命令

#镜像
docker pull xxx #从dockerhub拉取镜像
docker search xxx #在dockerhub查看镜像有版本
docker rmi 镜像名 #删除镜像
#容器
docker start 容器名 #启动容器
docker stop 容器名 #停止
docker rm 容器名 #删除容器
docker restart 容器名 #重启容器
docker inspect 容器id  #查看容器内部细节
docker top 容器id  #查看容器内运行的进程
docker cp 容器id:容器内路径 目标主机路径 #复制容器内文件到外部
docker exec -it 容器名 bash #进入容器内部，并且exit退出时保持容器运行

#打包以及生成镜像
#=================================================================================================
docker export 容器id > 名字.tar #导出容器为一个tar包，对应import
cat 名字.tar | docker import - 自定义用户名/镜像名:镜像版本号 #将一个export导出的tar包转换为镜像

docker commit -a="作者信息" -m="提交的描述信息" 容器id 容器名:版本号 #提交容器镜像为一个镜像

docker save -o xxx.tar 镜像名 #将指定镜像导出为一个tar包。
docker load -i xxx.tar #解析tar包到本地镜像库

#export和commit、save的区别。
#export只会保留最新的一层docker镜像层，无法做到回滚，优点是导出后的镜像小。常用于构建基础镜像
#commit会在docker镜像层中添加当前容器作为新的镜像层，并保留原有的读写层，可以根据镜像层回滚。常用于保存容器镜像
#save 会保留原有的读写镜像层，可以回滚，常用于备份镜像
#总结：export不能根据镜像层回滚，commit和save可以进行回滚，但是镜像大小比export大。
#=================================================================================================
systemctl enable docker //开机自启

2.搭建私有镜像仓库

docker pull registry 	#拉取私有仓库镜像
docker run --name registry -p5000:5000 -d -v /mydata/registry/:/tmp/registry --privileged=true registry #运行
docker tag redis 172.31.171.217:5000/redis:1.0 #打个标签将镜像更改为指定格式，docker tag 要上传的镜像:版本号 本机host:5000/自定义名:版本号
#docker默认不允许使用https的方式添加镜像，通过配置取消这个限制
#===========
vim /etc/docker/daemon.json 
{
"registry-mirrors":["https://aa25jngu..mirror.aliyuncs.com"],#配置源
"insecure-registries":["172.31.171.217:5000"] #添加可信任地址
}
systemctl restart docker 重启docker
#=============
docker push 172.31.171.217:5000/redis:1.0  #推送，已经打包好了标签直接 docker push 加镜像名即可
curl -XGET http://172.31.171.217:5000/v2/_catalog #查看仓库是否推送成功
docker pull 172.31.171.217:5000/redis:1.0  #拉取

3.示例

3.1mysql

docker run -p 3306:3306 --name mysql \
--privileged=true \
-v /mydata/mysql/log:/var/log/mysql \
-v /mydata/mysql/data:/var/lib/mysql \ 
-v /mydata/mysql/conf:/etc/mysql \ 
-e MYSQL_ROOT_PASSWORD=123456 \
-d mysql:5.7

#参数说明======
-p 3307:3306：将容器的 3306 端口映射到主机的 3307 端口
-v /mydata/mysql/conf:/etc/mysql：将配置文件夹挂在到主机
-v /mydata/mysql/log:/var/log/mysql：将日志文件夹挂载到主机
-v /mydata/mysql/data:/var/lib/mysql/：将数据文件夹挂载到主机，这个数据就是表和库
-e MYSQL_ROOT_PASSWORD=123456：初始化 root的密码为123456
#============

#解决中文乱码====
#在映射文件中conf目录中，创建一个配置文件，比如my.cnf。最好先创建好了文件，在让容器运行
vim my.cnf
#输入
[mysqld]
character-set-server=utf8 
[client]
default-character-set=utf8 
[mysql]
default-character-set=utf8
#==============

3.2主从复制版mysql

主节点容器配置

docker run -p 3307:3306 --name mysql-master \
--privileged=true \
-v /mydata/mysql/master/log:/var/log/mysql \
-v /mydata/mysql/master/data:/var/lib/mysql \
-v /mydata/mysql/master/conf:/etc/mysql \
-e MYSQL_ROOT_PASSWORD=123456 \
-d mysql:5.7

修改 master 基本配置
vim /mydata/mysql/master/conf/my.cnf
[client]
default-character-set=utf8
[mysql]
default-character-set=utf8
[mysqld]
init_connect='SET collation_connection = utf8_unicode_ci' 
init_connect='SET NAMES utf8' 
character-set-server=utf8
collation-server=utf8_unicode_ci
skip-character-set-client-handshake
skip-name-resolve
#注意：skip-name-resolve 一定要加，不然连接 mysql 会超级慢
#添加 master 主从复制部分配置
server_id=1 #server_id,同一局域网需要唯一
log-bin=mysql-bin #开启二进制日志功能
read-only=0 #
binlog_format=mixed #指定二进制格式 
expire_logs_days=7 #二进制日志过期时间，默认为0表示不自动清理
#binlog-do-db=lfy #可以指定需要同步的库，也可以直接不指定
slave_skip_errors=1062 #跳过主从复制中遇到的所有错误或指定类型错误，避免slave复制中断，1062是主键重复错误
#不同步的库
replicate-ignore-db=mysql 
replicate-ignore-db=sys
replicate-ignore-db=information_schema
replicate-ignore-db=performance_schema

创建授权连接的账号


docker exec -it mysql /bin/bash
#2、进入 mysql 内部 （mysql –uroot -p）
#1）、授权 root 可以远程访问（ 主从无关，为了方便我们远程连接 mysql）
grant all privileges on *.* to 'root'@'%' identified by '123456' with grant option;
flush privileges;
#2）、添加用来同步的用户
GRANT REPLICATION SLAVE ON *.* to 'lfyslave'@'%' identified by '123456';
#3、查看 master 状态
show master status; 
#记住这个状态的日志文件，需要给从节点进行指定日志，从节点根据日志来复制主节点的数据

配置 slaver 同步 master 数据

#1、进入 slaver 容器
docker exec -it mysql-slaver-01 /bin/bash
#2、进入 mysql 内部（mysql –uroot -p）
#1）、授权 root 可以远程访问（ 主从无关，为了方便我们远程连接 mysql）
grant all privileges on *.* to 'root'@'%' identified by '123456' with grant option;
flush privileges;

#2）、设置主库连接。master_connect_retry连接重试(秒),master_log_file从节点根据哪个日志进行复制，master_log_pos从节点从哪里开始复制
change master to master_host='ip',master_user='lfyslave',master_password='123456',master_log_file='mysql-bin.000001',master_log_pos=0,master_port=3307,master_connect_retry=30;

#3）、启动从库同步
start slave;

#4）、查看从库状态
show slave status\G

常见问题，当在slave中进行写操作时会造成数据不一致导致的错误

#1、先停掉slave
mysql> stop slave;
#2、跳过错误步数，后面步数可变
mysql> set global sql_slave_skip_counter=1;
#3、再启动slave
mysql> start slave;
#4、查看同步状态
mysql> show slave status\G;

最好创建一个只有读权限的用户用于连接从节点

#1、进入 slaver 容器
docker exec -it mysql-slaver-01 /bin/bash
#2、进入 mysql 内部（mysql –uroot -p）
create user 'lfy'@'%' identified by '123456';
grant select on *.* to 'lfy'@'%'; #给lfy用户select的权限，%表示任意ip可连接 *.*表示给于所有库中所有表，如果指定表例如可以指定mysql.user表
 grant SHOW VIEW on mysql.user to 'lfy'@'%';

#1）、授权 lfy 可以远程访问
grant all privileges on *.* to 'lfy'@'%' identified by '123456' with grant option;
flush privileges;
