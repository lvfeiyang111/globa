1.Spring中Aop校验JWT

//注解
@Target(ElementType.METHOD)
@Retention(RetentionPolicy.RUNTIME)
public @interface CheckJWT {

}
//切面类
@Aspect
@Component
@Slf4j
public class Accpet {
    @Pointcut("@annotation(com.lfy.aop.CheckJWT)")
    public void easyPuct(){
    }
    @Around("easyPuct()")
    public Object checkJWT(ProceedingJoinPoint point){
        ServletRequestAttributes requestAttributes = (ServletRequestAttributes) RequestContextHolder.getRequestAttributes();
        HttpServletRequest request = requestAttributes.getRequest();
        String jwt = request.getHeader("jwt");
        //伪代码，验证JWT
        if("1".equals(jwt)){
            try {
                return point.proceed();
            } catch (Throwable e) {
                log.error("验证JWT异常，异常:{}，方法：{}，入参：{}，JWT:{}",e.getMessage(),point.getSignature(),point.getArgs(),jwt);
                return "验证失败";
            }
        }else {
            return "jwt错误";
        }

    }
}

2.Spring中注解给方法参数赋值

1.第一步创建注解

@Retention(RetentionPolicy.RUNTIME)
@Target(ElementType.PARAMETER)
public @interface CheckUser {
}

2.第二步创建配置类

  • 实现HandlerMethodArgumentResolver
    • 这个是spring提供给参数进行校验的方法
    • supportsParameter方法如果为true就会进入resolveArgument方法中
      • parameter.getParameterType()：获取参数类型
      • parameter.hasParameterAnnotation(xxx.class) 判断是否包含注解类
    • resolveArgument方法的返回值会自动赋值给参数
      • 强转req HttpServletRequest req = (HttpServletRequest) webRequest.getNativeRequest();
public class CheckUserConfig implements HandlerMethodArgumentResolver {
    @Resource
    private RedisTemplate redisTemplate;
    @Resource
    private AdminMapper adminMapper;
    @Override
    public boolean supportsParameter(MethodParameter parameter) {
        //验证注解是否是Admin类
        boolean AdminCheck = parameter.getParameterType().equals(Admin.class);
		//验证注解是否有@CheckUser 注解
        boolean hasAnnotation = parameter.hasParameterAnnotation(CheckUser.class);
        //如果有，进入逻辑方法
        if (AdminCheck&&hasAnnotation){
            return true;
        }

        return false;
    }

    @Override
    public Object resolveArgument(MethodParameter parameter, ModelAndViewContainer mavContainer, NativeWebRequest webRequest, WebDataBinderFactory binderFactory) throws Exception {
        //强转req
        HttpServletRequest req = (HttpServletRequest) webRequest.getNativeRequest();
		//获取token
        String authorization = req.getHeader("Authorization");
        ValueOperations redis = redisTemplate.opsForValue();
        //通过redis获取id
        Integer adminid = (Integer) redis.get("Bearer" + authorization);
        //通过id查数据库找到admin,返回给使用注解的参数赋值
        Admin admin = adminMapper.selectOne(new QueryWrapper<Admin>().eq("id", adminid));
        return admin;
    }
}

3.把配置类注入到webmvcconfig

把config注入进bean，实现WebMvcConfigurer，重写addArgumentResolvers方法，把配置项添加进addArgumentResolvers

@Configuration
public class CheckUserResolverConfig implements WebMvcConfigurer {
    @Bean
public CheckUserConfig config(){
    return new CheckUserConfig();
}
    @Override
    public void addArgumentResolvers(List<HandlerMethodArgumentResolver> resolvers) {
        resolvers.add(config());
    }
}

3.使用工厂+策略消除 if-else

例如 根据参数1-7返回星期几

public String getDayNow(String code){
	if(code.equels("1")){
		return "星期一"
	}else if(code.equels("2")){
		return "星期二"
	}else if(code.equels("3")){
		return "星期三"
	} //等等等....省略
}

当方法需要扩展时，需要变动代码，太多的if判断分支导致代码犹如屎山一般。

解决：

//定义策略接口
public interface DayParse {
    //判断是否进入此策略
    Boolean Check(String code);
    //策略逻辑
    String service();
    //策略名
    String GetName();
    //优先级
    int GetPriority();
}

定义星期一类

public class Monday implements DayParse {
    @Override
    public Boolean Check(String code) {
        if(code.equals("1")){
            return true;
        }
        return false;
    }
    @Override
    public String service() {
        return "星期一";
    }
    @Override
    public String GetName() {
        return "1";
    }
    @Override
    public int GetPriority() {
        return 1;
    }
}

定义星期二

public class Tuesday implements DayParse {
    @Override
    public Boolean Check(String code) {
        if(code.equals("2")){
            return true;
        }
        return false;
    }

    @Override
    public String service() {
        return "星期二";
    }

    @Override
    public String GetName() {
        return "1";
    }

    @Override
    public int GetPriority() {
        return 1;
    }
}

以此类推，定义7个实现类

定义工厂

public class MonFactoryBean {
    private static MonFactoryBean bean = new MonFactoryBean();
    private static List<MonFactorParse> list = new ArrayList<>();
    private MonFactorBean(){
        //将实现类实例化添加进list
        list.add(new Monday());
        list.add(new Tuesday());
        list.add(new Wednesday());
        //省略.......
    }
    //作为工厂单例无需多言
    public static MonFactorBean getInstence(){
        return bean;
    }
    //返回实现类
    public DayParse getDay(String code){
        DayParse p = null;
        for (MonFactorParse monFactorParse : list) {
            //遍历出符合策略的方法返回
            if(monFactorParse.Check(code)){
                //校验优先级，（这里的误区，很多人以为会空指针）当p不为空时，才会走到第二个优先级判断，所以不会出现空指针
                if(p == null || monFactorParse.GetPriority() > p.GetPriority()){
                    p = monFactorParse;
                }

            }
        }
        if(p==null){
            throw new RuntimeException("没有找到策略");
        }
        return p;
    }

}

4.使用责任链消除 if-else

场景：现在有员工需要请假，根据请假天数的不同，需要不同负责人的审批，那么如何使用责任链模式消除if-else呢

定义抽象类


public abstract class LeaveHanler {
    //下一个处理者
    protected LeaveHanler nextLeave;
    //处理请假天数范围
    protected Integer beginDays;
    protected Integer endDays;

    public LeaveHanler(Integer beginDays, Integer endDays) {
        this.beginDays = beginDays;
        this.endDays = endDays;
    }
    //处理请假逻辑
    public abstract void service();
    
    //提交
    public final void submit(People p){
        //如果符合当前小组长的处理权限则调用该组长的service
        if(p.getNum()<=this.endDays&&p.getNum()>=this.beginDays){
            this.service();
        }else if(this.nextLeave!=null){
            //否则交给下一个处理者的submit
            this.nextLeave.submit(p);
        }else {
            System.out.println("超出权限范围，没人能处理");
        }
    }

    public void setNextLeave(LeaveHanler nextLeave) {
        this.nextLeave = nextLeave;
    }
}

//请假实体
@Data
public class People {
    private String name;//姓名
    private int num;//请假天数
    private String content;//请假内容
}
//小组长1
public class Leader1 extends LeaveHanler{
    public Leader1() {
        super(1, 3);
    }

    @Override
    public void service() {
        System.out.println("小组长1处理");
    }
}
//小组长2
public class Leader2 extends LeaveHanler{
    public Leader2() {
        super(4, 6);
    }

    @Override
    public void service() {
        System.out.println("小组长2处理");
    }
}
//客户端类
public class client {
    public static void main(String[] args) {
        //这里可以抽取代码
        Leader1 leader1 = new Leader1();
        Leader2 leader2 = new Leader2();
        leader1.setNextLeave(leader2);
        People people = new People("张三", 4, "xxx");

        leader1.submit(people);

    }
}

6.定时任务

@EnableScheduling //开启定时任务
@Component
@Slf4j
public class SchedulingConfig {
    
    @Scheduled(cron = "* * * * * ?") //执行周期 
    public void test(){
        //最好配合线程池使用，这里省略
        log.info("定时任务执行ing...");
    }
}

  • 语法：秒 分 时 日 月 周 年（年Spring 不支持）

  • 特殊字符：

    ，：枚举；

    (cron=”7,9,23 * * * * ?”)：任意时刻的 7,9，23 秒启动这个任务；

    **-**：范围：

    (cron=”7-20 * * * * ?”):任意时刻的 7-20 秒之间，每秒启动一次

    *****：任意；

    指定位置的任意时刻都可以

    **/**：步长；

    (cron=”7/5 * * * * ?”)：第 7 秒启动，每 5 秒一次；

    (cron=”*/5 * * * * ?”)：任意秒启动，每 5 秒一次；

    ？：（出现在日和周几的位置）：为了防止日和周冲突，在周和日上如果要写通配符使

    用?

    (cron=”* * * 1 * ?”)：每月的 1 号，启动这个任务；

    L：（出现在日和周的位置）”，

    last：最后一个

    (cron=”* * * ? * 3L”)：每月的最后一个周二

    W：

    Work Day：工作日

    (cron=”* * * W * ?”)：每个月的工作日触发

    (cron=”* * * LW * ?”)：每个月的最后一个工作日触发

    **#**：第几个

    (cron=”* * * ? * 5#2”)：每个月的第 2 个周 4
